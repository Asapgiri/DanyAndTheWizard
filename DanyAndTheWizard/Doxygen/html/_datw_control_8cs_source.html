398  col.Width = new GridLength(90);

403  grid.ColumnDefinitions.Add(col);

423  btn.Content = "Use";
424  btn.Click += (sender, e) =>
425  {
426  this.Logic.UseItem(item);

428  };
429  }

431  {
432  btn.Content = "Unequip";

434  {
435  this.Logic.UnEquipItem(item);
436  this.InventoryOpen();
437  };

439  }
440  else

442  btn.Content = "Equip";
443  btn.Click += (sender, e) =>
444  {
445  this.Logic.EquipItem(item);
446  this.InventoryOpen();
447  };
448  }
449 
450  btn.Margin = new Thickness(5);

531  }
