282  private Drawing DrawMyText(string textString, Point where, FlowDirection fdr, int fontSize, Brush background, bool setWidth = false, double opi = 1, bool middle = false)
283  {
284  // Create a new DrawingGroup of the control.
285  DrawingGroup drawingGroup = new DrawingGroup();
286 
287  // Open the DrawingGroup in order to access the DrawingContext.
288  using (DrawingContext drawingContext = drawingGroup.Open())
289  {
290  // Create the formatted text based on the properties set.
291  FormattedText formattedText = new FormattedText(
292  textString,
293  CultureInfo.GetCultureInfo("en-us"),
294  fdr,
295  new Typeface("Comic Sans MS Bold"),
296  fontSize,
297  System.Windows.Media.Brushes.Black); // This brush does not matter since we use the geometry of the text.
298 
299  if (setWidth)
300  {
301  formattedText.MaxTextWidth = this.model.GameWidth - where.X;
302  formattedText.MaxTextHeight = 300;
303  }
304 
305  Geometry textGeometry;
306  if (middle)
307  {
308  textGeometry = formattedText.BuildGeometry(new Point(where.X - (formattedText.Width / 2), where.Y));
309  if (fdr == FlowDirection.LeftToRight)
310  {
311  drawingContext.DrawRoundedRectangle(background, null, new Rect(where.X - (formattedText.Width / 2) - 25, where.Y, formattedText.Width + 50, formattedText.Height + 5), 7.5, 7.5);
312  }
313  }
314  else
315  {
316  // Build the geometry object that represents the text.
317  textGeometry = formattedText.BuildGeometry(where);
318  if (fdr == FlowDirection.LeftToRight)
319  {
320  drawingContext.DrawRoundedRectangle(background, null, new Rect(where.X - 25, where.Y, formattedText.Width + 50, formattedText.Height + 5), 7.5, 7.5);
321  }
322  }
323 
324  // Draw the outline based on the properties that are set.
325  drawingContext.DrawGeometry(System.Windows.Media.Brushes.Gold, new System.Windows.Media.Pen(System.Windows.Media.Brushes.Maroon, 1.5), textGeometry);
326  }
327 
328  drawingGroup.Opacity = opi;
329 
330  // Return the updated DrawingGroup content to be used by the control.
331  return drawingGroup;
332  }

358  this.model.Map[(int)this.model.ActualMapSegment.X, (int)this.model.ActualMapSegment.Y].Entities[i, j].Name.Split('|')[0] != "NextMap" &&
